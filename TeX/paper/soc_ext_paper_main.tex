\begin{abstract}\noindent 
	\footnotesize{\begin{center}\textbf{Abstract}\end{center} 
		This paper investigates the impact of professional football games on physical violence in Germany. I employ a generalized difference-in-differences strategy that exploits time-series and cross-sectional variation in crime register data for the period 2011-2015. I compare the number of assaults on days with and without football matches, while controlling for date heterogeneity, weather, and holidays. My results suggest that a home game leads to a 21.5 percent increase in the assault rate. The effects are largely driven by male victims who do not know the perpetrator, and decrease with the victims' age. In total, professional football games explain almost 18 percent of all assault reports and generate annual social costs of 95 million euros.	\\\newline \textbf{Keywords:} football games, violent crime, assaults. \newline \textbf{JEL codes:} J19, K42, Z13, Z29.}

\end{abstract}

\section*{Idea collection}
\underline{Further Possible interactions:} 
\begin{itemize}
	\item $(\gamma_d \times \delta_m)$: month-specific DOW fixed effects
	\item $r \times time$: e.g. $r \times w$, $r \times y \times m$, $r \times y \times w$
\end{itemize}

\underline{Other approaches to compute standard errors:}
\begin{itemize}
	\item clustered at club level $c$ (allow for correlation over time within region and across regions corresponding to the same club)

	\item adjust for serial correlation: 
	\begin{itemize}
		\item BDM (2004) clustered at club level
		\item CGM (2008) wild cluster bootstrap t-procedures
	\end{itemize}
\end{itemize}

\underline{drop observations:}
\begin{itemize}
	\item focus on only season (i.e. exclude winter break)
	\item how many games are played on holidays? drop these? 
	\item drop when information of crime is not available on the hour
\end{itemize}
